arduinoのセンサー


arduinoでserial monitorの値が300を超えたときledライトが点滅するようにするコード
  1. const int ledPin = 13; // LEDを接続したデジタルピン番号
  2. int threshold = 300; // しきい値
  4. void setup() {
  5.   pinMode(ledPin, OUTPUT); // LEDのピンを出力モードに設定
  6.   Serial.begin(9600); // シリアル通信を開始（ボーレートは9600）
  7. }
  9. void loop() {
  10.   // シリアルポートからのデータを読み取る
  11.   int val=0;
  12.   val=analogRead(1);
  13.   // 読み取った値をシリアルモニターに表示
  14.   Serial.println(val);
  15.  
  16.     // もしセンサーの値がしきい値以上であればLEDを点滅させる
  17.     if (val >= threshold) {
  18.       digitalWrite(ledPin, HIGH); // LEDを点灯
  19.       delay(500); // 0.5秒待つ
  20.       digitalWrite(ledPin, LOW); // LEDを消灯
  21.       delay(500); // 0.5秒待つ
  22.     } else{
  23.       digitalWrite(ledPin, LOW); // LEDを消灯
  24.     }
  25.   
  26.    delay(200);
  27. }
